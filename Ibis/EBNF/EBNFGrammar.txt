
whitespace = ? whitespace characters ?;

character = ? a-z and A-Z ?;

number = ? 0-9 ?;



rule = [whitespace] name [whitespace] '=' rule_body ';';

name = character {character | number};

rule_body = [whitespace] rule_section {whitespace '|' whitespace rule_section} [whitespace];

rule_section = rule_statement {whitespace rule_statement};

rule_statement = optional | repetition | grouping | literal | name;



optional = '[' rule_body ']';

repetition = '{' rule_body '}';

grouping = '(' rule_body ')';

literal = ''' (character | number) {character | number} ''';
